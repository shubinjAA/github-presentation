  await page.goto('https://app.example.com/login');

  await page.getByLabel('Email').fill('user@example.com');
  await page.getByLabel('Password').fill('Test@1234');
  await page.getByRole('button', { name: 'Sign In' }).click();
  
  // Wait for navigation
  await page.waitForURL('**/dashboard');
  
  // Verify success

    .toBeVisible();
});

What the Agent Did


// 1. Retrieved user story from ADO
const story = await adoMcp.query(`
  SELECT * FROM WorkItems 
  WHERE Title CONTAINS 'Checkout Payment Flow'
`);

// 2. Generated test scenarios from acceptance criteria
const scenarios = await agent.parseAcceptanceCriteria(story);
// 20 scenarios identified including edge cases

// 3. Created Playwright tests
for (const scenario of scenarios) {
  await playwrightMcp.generateTest({
    name: scenario.name,
    steps: scenario.steps,
    assertions: scenario.expectedResults
  });
}

// 4. Executed across Chrome, Firefox, Safari
const results = await playwrightMcp.runTests({ browsers: 'all' });

// 5. Updated ADO test results
await adoMcp.updateTestResults(story.id, results);
